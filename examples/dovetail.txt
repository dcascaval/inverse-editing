// Dovetail joint: two interlocking plates with trapezoidal fingers.

parameters {
  baseW: 40
  baseH: 6
  fingerH: 3 < 6 < 12
  taper: 0 < 15 < 40
  gap: 0 < 0.5 < 2
  chmf: 0 < 1 < 3
}

nFingers = 3
fingerW = baseW / (2 * nFingers + 1)
dx = fingerH * tan(taper * PI / 180)

// Tail board: base plate with protruding trapezoidal fingers
tailPlate = rect(0, 0, baseW, baseH)

operation finger(cx) {
  body = rect(cx - fingerW/2, baseH, fingerW, fingerH)
  leftWedge = rect(cx - fingerW/2 - dx, baseH, dx, fingerH)
  rightWedge = rect(cx + fingerW/2, baseH, dx, fingerH)
  shape = unionAll(body, leftWedge, rightWedge)
}

f0 = finger(fingerW * 1.5)
f1 = finger(fingerW * 3.5)
f2 = finger(fingerW * 5.5)

tail = unionAll(tailPlate, f0.shape, f1.shape, f2.shape)

draw(tail, translucent(0.2))
draw(tail.edges, color(gray))

// Pin board: base plate offset above, with mating slots
pinY = baseH + fingerH + gap
pinPlate = rect(0, pinY, baseW, baseH)

// Mating fingers point downward (mirrored)
mirrorLine = line(pt(0, pinY), pt(1, pinY))
g0 = finger(fingerW * 0.5)
g1 = finger(fingerW * 2.5)
g2 = finger(fingerW * 4.5)

p0 = g0.shape.mirror(mirrorLine)
p1 = g1.shape.mirror(mirrorLine)
p2 = g2.shape.mirror(mirrorLine)

pin = unionAll(pinPlate, p0, p1, p2)

draw(pin, translucent(0.2))
draw(pin.edges, color(blue))
